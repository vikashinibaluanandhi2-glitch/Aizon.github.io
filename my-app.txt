<script>
  const USD_STAKED = 15295272.69;
  const USD_MARKET = 26689575.64;
  const REFRESH_INTERVAL = 30; /* seconds */

  let countdown = REFRESH_INTERVAL;
  let intervalId = null;
  let countdownId = null;
  let lastRate = null;

  function fmtINR(n) {
    if (n >= 1e7) return '₹' + (n / 1e7).toFixed(2) + ' Cr';
    if (n >= 1e5) return '₹' + (n / 1e5).toFixed(2) + ' L';
    return '₹' + n.toLocaleString('en-IN', { maximumFractionDigits: 0 });
  }

  function fmtUSD(n) {
    return '$' + n.toLocaleString('en-US', { maximumFractionDigits: 2 });
  }

  function flash(id) {
    const el = document.getElementById(id);
    if (!el) return;
    el.classList.remove('stat-flash');
    void el.offsetWidth;
    el.classList.add('stat-flash');
  }

  function setValues(rate, isLive) {
    const staked = USD_STAKED * rate;
    const market = USD_MARKET * rate;

    if (lastRate !== rate) {
      ['staked','market','rate'].forEach(flash);
    }
    lastRate = rate;

    document.getElementById('staked').textContent = fmtINR(staked);
    document.getElementById('staked-usd').textContent = fmtUSD(USD_STAKED) + ' USD';
    document.getElementById('market').textContent = fmtINR(market);
    document.getElementById('market-usd').textContent = fmtUSD(USD_MARKET) + ' USD';
    document.getElementById('rate').textContent = '₹' + rate.toFixed(2);
    document.getElementById('rate-time').textContent = isLive
      ? 'Updated ' + new Date().toLocaleTimeString('en-IN')
      : 'Estimated · live rate unavailable';

    document.getElementById('footer-status').textContent = isLive
      ? 'Live · next update in ' + countdown + 's'
      : 'Offline · retrying...';

    document.querySelectorAll('.stat-loading').forEach(el => el.classList.remove('stat-loading'));
  }

  async function fetchAll() {
    try {
      const res = await fetch('https://api.frankfurter.app/latest?from=USD&to=INR');
      if (!res.ok) throw new Error('bad response');
      const data = await res.json();
      setValues(data.rates.INR, true);
      resetCountdown();
    } catch (e) {
      setValues(lastRate || 83.5, false);
    }
  }

  function resetCountdown() {
    countdown = REFRESH_INTERVAL;
    updateBars(1.0);
  }

  function updateBars(fraction) {
    document.querySelectorAll('.countdown-bar').forEach(bar => {
      bar.style.width = (fraction * 100) + '%';
    });
  }

  function startCountdown() {
    clearInterval(countdownId);
    countdownId = setInterval(() => {
      countdown = Math.max(0, countdown - 1);
      const fraction = countdown / REFRESH_INTERVAL;
      updateBars(fraction);
      const statusEl = document.getElementById('footer-status');
      if (statusEl && lastRate) {
        statusEl.textContent = 'Live · next update in ' + countdown + 's';
      }
    }, 1000);
  }

  function addCountdownBars() {
    document.querySelectorAll('.stat-card').forEach(card => {
      if (!card.querySelector('.countdown-bar')) {
        const bar = document.createElement('div');
        bar.className = 'countdown-bar';
        bar.style.width = '100%';
        card.appendChild(bar);
      }
    });
  }

  /* Pause updates when tab is hidden, resume when visible */
  document.addEventListener('visibilitychange', () => {
    if (document.hidden) {
      clearInterval(intervalId);
      clearInterval(countdownId);
    } else {
      fetchAll();
      intervalId = setInterval(fetchAll, REFRESH_INTERVAL * 1000);
      startCountdown();
    }
  });

  /* Init */
  addCountdownBars();
  fetchAll();
  intervalId = setInterval(fetchAll, REFRESH_INTERVAL * 1000);
  startCountdown();
</script>
